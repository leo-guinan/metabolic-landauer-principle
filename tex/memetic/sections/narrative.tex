\section{Narrative Hooks and Meme Bundles}

\subsection{Core Memes}
\begin{itemize}
  \item \textbf{Delete to Heal}: clarity frees energy.
  \item \textbf{Entropy of Emotion}: unresolved loops weigh down metabolism.
  \item \textbf{Compression = Insight}: the ``aha'' moment as thermodynamic efficiency.
  \item \textbf{Trust Releases Energy}: social confidence lowers deletion cost.
\end{itemize}

\subsection{Story Starters}
\begin{itemize}
  \item ``The physics of letting go: last night I deleted an unresolved apology and woke up with a 10-point HRV jump.''
  \item ``Emotional fat is unprocessed data: here is what happened when I compressed my career regret into one sentence.''
  \item ``Can understanding burn calories? My smartwatch says maybe.''
  \item ``Every insight is a tiny nuclear reaction---energy released when confusion collapses.''
\end{itemize}

\subsection{Visual Templates}
\begin{itemize}
  \item Before/after thermodynamic meters showing $\uncertainty/\confidence$ ratios.
  \item Flow charts mapping loop \textrightarrow{} compression \textrightarrow{} deletion \textrightarrow{} energy.
  \item Overlay graphs of HRV and self-reported clarity.
\end{itemize}
